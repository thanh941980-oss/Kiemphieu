async function loadChungKien() { try { console.log('Loading chungKien data...'); const userPath = getUserPath('chungKien'); console.log('User path:', userPath); const tableBody = document.querySelector('#tableChungKien tbody'); tableBody.innerHTML = ''; const snapshot = await db.ref(userPath).once('value'); console.log('Snapshot exists:', snapshot.exists()); if (snapshot.exists()) { console.log('Data found, processing...'); snapshot.forEach(child => { const data = child.val(); console.log('ChungKien item:', data); const row = document.createElement('tr'); row.innerHTML = ` ${data.hoTen} ${data.diaChi} `; tableBody.appendChild(row); }); } else { console.log('No data found for chungKien'); tableBody.innerHTML = 'Chưa có dữ liệu'; } } catch (error) { console.error('Error loading chungKien:', error); document.querySelector('#tableChungKien tbody').innerHTML = 'Lỗi tải dữ liệu'; } }
